Mastering E2E Testing

Test Complete User Workflows

What is E2E Testing?

End-to-End testing verifies complete user workflows from start to finish, testing the entire application stack including frontend, backend, database, and external services.

Playwright Example

import { test, expect } from '@playwright/test';

test.describe('User Authentication', () => {
    test('user can login successfully', async ({ page }) => {
        // Navigate to login page
        await page.goto('https://example.com/login');
        
        // Fill in credentials
        await page.fill('[name="email"]', 'user@example.com');
        await page.fill('[name="password"]', 'password123');
        
        // Submit form
        await page.click('button[type="submit"]');
        
        // Wait for navigation to dashboard
        await page.waitForURL('**/dashboard');
        
        // Verify user is logged in
        await expect(page.locator('h1')).toContainText('Dashboard');
        await expect(page.locator('.user-name')).toContainText('John Doe');
        
        // Take screenshot
        await page.screenshot({ path: 'dashboard.png' });
    });
    
    test('shows error for invalid credentials', async ({ page }) => {
        await page.goto('https://example.com/login');
        
        await page.fill('[name="email"]', 'wrong@example.com');
        await page.fill('[name="password"]', 'wrongpass');
        await page.click('button[type="submit"]');
        
        // Verify error message
        await expect(page.locator('.error-message'))
            .toContainText('Invalid credentials');
    });
});

Cypress Example

describe('Shopping Cart Flow', () => {
    beforeEach(() => {
        cy.visit('/');
    });
    
    it('completes full purchase flow', () => {
        // Browse products
        cy.get('[data-testid="products-link"]').click();
        cy.url().should('include', '/products');
        
        // Add item to cart
        cy.get('[data-testid="product-1"]').within(() => {
            cy.contains('Add to Cart').click();
        });
        
        // Verify cart badge updated
        cy.get('[data-testid="cart-badge"]').should('contain', '1');
        
        // Go to cart
        cy.get('[data-testid="cart-icon"]').click();
        
        // Verify cart contents
        cy.get('.cart-item').should('have.length', 1);
        cy.get('.cart-total').should('contain', '$29.99');
        
        // Proceed to checkout
        cy.get('[data-testid="checkout-btn"]').click();
        
        // Fill shipping info
        cy.get('[name="fullName"]').type('John Doe');
        cy.get('[name="address"]').type('123 Main St');
        cy.get('[name="city"]').type('New York');
        cy.get('[name="zipCode"]').type('10001');
        
        // Fill payment info
        cy.get('[name="cardNumber"]').type('4242424242424242');
        cy.get('[name="expiry"]').type('12/25');
        cy.get('[name="cvv"]').type('123');
        
        // Submit order
        cy.get('[data-testid="submit-order"]').click();
        
        // Verify success
        cy.url().should('include', '/order-confirmation');
        cy.contains('Order confirmed').should('be.visible');
        cy.get('[data-testid="order-number"]').should('exist');
    });
    
    it('allows removing items from cart', () => {
        // Add items
        cy.visit('/products');
        cy.get('[data-testid="product-1"]').click();
        cy.get('[data-testid="add-to-cart"]').click();
        
        // Go to cart
        cy.get('[data-testid="cart-icon"]').click();
        
        // Remove item
        cy.get('[data-testid="remove-item"]').click();
        
        // Verify cart is empty
        cy.contains('Your cart is empty').should('be.visible');
    });
});

Handling Async Operations

// Playwright - automatic waiting
test('waits for elements', async ({ page }) => {
    await page.goto('/dashboard');
    
    // Automatically waits for element to be visible
    await page.click('button');
    
    // Wait for specific condition
    await page.waitForSelector('.data-loaded');
    
    // Wait for network request
    await page.waitForResponse(response =>
        response.url().includes('/api/data') && response.status() === 200
    );
});

// Cypress - built-in retry logic
it('waits for elements', () => {
    cy.visit('/dashboard');
    
    // Automatically retries until element exists
    cy.get('.loading').should('not.exist');
    cy.get('.data-loaded').should('be.visible');
    
    // Wait for API call
    cy.intercept('GET', '/api/data').as('getData');
    cy.wait('@getData');
});

Page Object Pattern

// pages/LoginPage.js
export class LoginPage {
    constructor(page) {
        this.page = page;
        this.emailInput = page.locator('[name="email"]');
        this.passwordInput = page.locator('[name="password"]');
        this.submitButton = page.locator('button[type="submit"]');
        this.errorMessage = page.locator('.error-message');
    }
    
    async goto() {
        await this.page.goto('/login');
    }
    
    async login(email, password) {
        await this.emailInput.fill(email);
        await this.passwordInput.fill(password);
        await this.submitButton.click();
    }
    
    async getErrorMessage() {
        return await this.errorMessage.textContent();
    }
}

// test file
import { LoginPage } from './pages/LoginPage';

test('login with page object', async ({ page }) => {
    const loginPage = new LoginPage(page);
    
    await loginPage.goto();
    await loginPage.login('user@example.com', 'password123');
    
    await expect(page).toHaveURL('**/dashboard');
});

Best Practices

  • Test critical paths - Focus on important user journeys
  • Use data-testid - Stable selectors independent of styling
  • Keep tests independent - Each test should run in isolation
  • Handle waits properly - Use framework's built-in waiting
  • Use page objects - Encapsulate page interactions
  • Run in CI/CD - Automate E2E tests in pipeline
  • Avoid flaky tests - Use reliable selectors and waits
  • Keep tests fast - Parallelize when possible

Key Takeaways

  • E2E tests verify complete user workflows
  • Playwright and Cypress are popular frameworks
  • Test from user's perspective
  • Use data-testid for stable selectors
  • Handle async operations with built-in waits
  • Page object pattern improves maintainability
  • Run regularly in CI/CD pipeline
  • Catch integration and workflow issues
